  7class ExprNode:
  8    """
  9    Class representing a binary Expression Tree in form of a recursive node.
 10
 11    Attributes
 12    ----------
 13    data: str
 14        This node content, may be a number or an algebraic operator.
 15    node_left: ExprNode
 16        The left sub-node, if empty this node is a leaf.
 17    node_right: ExprNode
 18        The right sub-node, if empty this node is a leaf.
 19
 20    Methods
 21    ----------
 22    is_leaf()
 23        Procedure to check whether the node is a leaf or not.
 24    as_prefix()
 25        Procedure to create the prefix string describing the tree.
 26    as_infix()
 27        Procedure to create the infix string describing the tree.
 28    get_disjunctions_infix()
 29        Procedure to return the list of disjunct atoms, if present.
 30
 31    """
 32
 33    def __init__(self, char: str):
 34        self.data = char
 35        self.node_left = None
 36        self.node_right = None
 37
 38    def is_leaf(self) -> bool:
 39        """
 40        This method checks whether the node is a leaf or not.
 41
 42        Returns
 43        ----------
 44        bool
 45            True if node_left and node_right are None, False otherwise.
 46
 47        """
 48
 49        return self.node_left is None and self.node_right is None
 50
 51    def as_prefix(self) -> str:
 52        """
 53        This method converts the Expression Tree in a prefix string
 54
 55        Returns
 56        ----------
 57        str
 58            The tree representation as a prefix string
 59
 60        """
 61
 62        if self.data.startswith('-'):
 63            data = self.data.replace('-', '(- ')
 64            self.data = data + ')'
 65
 66        if self.node_left is None:
 67            return self.data
 68        else:
 69            prefix = '(' + self.data
 70            prefix += ' ' + self.node_left.as_prefix()
 71            prefix += ' ' + self.node_right.as_prefix()
 72            return prefix + ')'
 73
 74    def as_infix(self) -> str:
 75        """
 76        This method converts the Expression Tree in an infix string
 77
 78        Returns
 79        ----------
 80        str
 81            The tree representation as an infix string
 82
 83        """
 84
 85        if self.node_left is None:
 86            return self.data
 87        else:
 88            infix = '(' + self.node_left.as_infix()
 89            infix += ' ' + self.data
 90            infix += ' ' + self.node_right.as_infix()
 91            return infix + ')'
 92
 93    def get_disjunctions_infix(self) -> list:
 94        """
 95        This method is used in order to separate OR statements in
 96        the input file
 97
 98        Returns
 99        ----------
100        list
101            The list of infix-notated disjunctions
102
103        """
104
105        if self.data != '|':
106            raise Exception('No disjunctions detected')
107
108        d_list = [self.node_right.as_infix()]  # Right depth is 1
109        if self.node_left.data != '|':
110            d_list.append(self.node_left.as_infix())
111        else:
112            d_list.extend(self.node_left.get_disjunctions_infix())
113
114        return d_list

117class ExpressionTreeConverter:
118    """
119    Class for converting infix expressions to expression trees.
120
121    Courtesy of Nikhil Kumar Singh(nickzuck_007) and aashish1995.
122
123    Attributes
124    ----------
125    precedence: dict
126        A dictionary assigning precedences to operators.
127    charStack: list
128        Stack for operands, needed in the conversion routine.
129    nodeStack: list
130        Stack for operators, needed in the conversion routine.
131
132    Methods
133    ----------
134    build_from_infix(str)
135        Procedure to generate an expression tree from an infix string.
136
137    """
138
139    def __init__(self):
140        self.precedence = {'+': 1, '-': 1, '*': 2, '/': 2, '>=': 3, '<=': 3, '>': 3, '<': 3, '=': 3, '&': 4, '|': 4}
141        self.charStack = []
142        self.nodeStack = []
143
144    def build_from_infix(self, infix: str) -> ExprNode:
145        """
146        This method builds an Expression Tree using the ExprNode class.
147
148        Parameters
149        ----------
150        infix: str
151            The infix-notated string to be converted.
152
153        Returns
154        ----------
155        ExprNode
156            The root node containing the Expression Tree.
157
158        """
159
160        infix = '(' + infix + ')'  # Redundancy for correctness
161        # Expand parentheses in order to split correctly
162        infix = infix.replace('(', '( ').replace(')', ' )')
163
164        # Read tokens
165        for char in infix.split():
166            # Expression beginning
167            if char == '(':
168                self.charStack.append(char)
169
170            # Expression end
171            elif char == ')':
172                while len(self.charStack) > 0 and self.charStack[-1] != '(':
173                    t = ExprNode(self.charStack.pop())
174                    t1 = self.nodeStack.pop()
175                    t2 = self.nodeStack.pop()
176
177                    t.node_left = t2
178                    t.node_right = t1
179                    self.nodeStack.append(t)
180
181                self.charStack.pop()
182
183            # Operand
184            elif not is_operator(char):
185                t = ExprNode(char)
186                self.nodeStack.append(t)
187
188            # Operator
189            elif self.precedence[char] > 0:
190                while len(self.charStack) > 0 and self.charStack[-1] != '(' and \
191                        self.precedence[self.charStack[-1]] >= self.precedence[char]:
192                    # Assign nodes
193                    t = ExprNode(self.charStack.pop())
194                    t1 = self.nodeStack.pop()
195                    t2 = self.nodeStack.pop()
196
197                    # Link left and right nodes
198                    t.node_left = t2
199                    t.node_right = t1
200                    self.nodeStack.append(t)
201
202                self.charStack.append(char)
203
204        return self.nodeStack.pop()

307    def __get_coef_mat(self, vector: list, vec_name: str, asserts: list) -> Tensor:
308        """
309        This method reads the assertions and extracts the coefficients
310        for the given vector.
311
312        Parameters
313        ----------
314        vector: list
315            The vector of interest.
316        vec_name: str
317            The vector name.
318        asserts: list
319            The list of SMTLIB 'assert' commands obtained via pysmt.
320
321        Returns
322        ----------
323        Tensor
324            The coefficient matrix associated to 'vec_name' and the
325            assertion list.
326
327        """
328
329        # Prepare output
330        n_var = len(vector)
331        coef_mat = np.zeros((len(asserts), n_var))
332
333        # Row counter
334        loop = 0
335
336        for a in asserts:
337            # Tokenize
338            line = a.replace('(', ' ( ').replace(')', ' ) ').split()
339            row = np.zeros(n_var)
340            idx = []
341
342            # Get variables index
343            for elem in line:
344                if vec_name in elem:
345                    idx.append(line.index(elem))
346
347            # Get coefficients for each index
348            for i in idx:
349                # Variable number --- corresponding to the vector index
350                v_num = int(line[i].replace(vec_name + '_', ''))
351                before = line[i - 1]
352
353                # var on lhs
354                if before == '(':
355                    # Check if the coefficient is on the right side
356                    if line[i + 1] == '*':
357                        row[v_num] += read_smt_num(line[i + 2])
358                    elif line[i + 1] == '/':
359                        row[v_num] += float(1 / read_smt_num(line[i + 2]))
360                    else:
361                        row[v_num] += 1
362
363                # var as addendum (c = 1)
364                elif before == '+':
365                    row[v_num] += 1
366
367                # Same, but negative (c = -1)
368                elif before == '-':
369                    row[v_num] -= 1
370
371                # var with coefficient
372                elif before == '*':
373                    row[v_num] += read_smt_num(line[i - 2])
374
375            # Row now is complete and can be put in the matrix
376            coef_mat[loop] = row
377            loop += 1
378
379        return coef_mat

381    def __get_bias_mat(self, asserts: list) -> Tensor:
382        """
383        This method reads the assertions and extracts the
384        known terms in order to build the bias vector.
385
386        Parameters
387        ----------
388        asserts: list
389            The list of SMTLIB 'assert' commands obtained via pysmt.
390            
391        Returns
392        ----------
393        Tensor
394            The bias matrix associated to the assertion list.
395
396        """
397
398        # Init
399        bias_mat = np.zeros((len(asserts), 1))
400        tree_converter = ExpressionTreeConverter()
401
402        # Row counter
403        loop = 0
404
405        for a in asserts:
406            # Tokenize
407            line = a.replace('(', ' ( ').replace(')', ' ) ')
408
409            b_leaf = tree_converter.build_from_infix(line).node_right
410            assert b_leaf.is_leaf(), 'The node' + b_leaf + 'was not leaf'
411
412            bias_mat[loop] = read_smt_num(b_leaf.data)
413            loop += 1
414
415        return bias_mat

417    @staticmethod
418    def remove_parentheses(cmd: str) -> str:
419        strip_idx = 0
420
421        # Forward
422        while cmd[strip_idx + 1] == '(':
423            strip_idx += 1
424
425        ret = cmd[strip_idx:]
426        strip_idx = len(ret)
427
428        # Backward
429        while ret[strip_idx - 1] == ')':
430            strip_idx -= 1
431
432        ret = ret[:strip_idx + 1]
433
434        return ret

465    def parse_property(self) -> (Tensor, Tensor, list, list):
466        """
467        This method exposes the propriety parsing, performing all the steps and
468        filling the Tensors.
469
470        Returns
471        ----------
472        NeVerProperty
473            The parsed property wrapped in the corresponding class
474
475        """
476
477        x_assert = self.__get_assert_commands_for(self.x_name)
478        y_assert = self.__get_assert_commands_for(self.y_name)
479
480        disjunct_list = []
481        tree_converter = ExpressionTreeConverter()
482
483        # If y_assert contains some 'OR' statement I should separate
484        for a in y_assert:
485            if '|' in a:
486                tree = tree_converter.build_from_infix(a)
487                disjunct_list.extend(tree.get_disjunctions_infix())
488
489        for i in range(len(disjunct_list)):
490            # Process 'and' in output property
491            if ' & ' in disjunct_list[i]:
492                disjunct_list[i] = disjunct_list[i].split(' & ')
493
494                count = 0
495                for a in disjunct_list[i]:
496                    disjunct_list[i][count] = self.remove_parentheses(a)
497                    count += 1
498
499        # Refine assert statements
500        for idx, a in enumerate(x_assert):
501            x_assert[idx] = refine_smt_statement(a, self.x_name)
502
503        if len(disjunct_list) == 0:
504            for idx, a in enumerate(y_assert):
505                y_assert[idx] = refine_smt_statement(a, self.y_name)
506        else:
507            for d_idx, statement in enumerate(disjunct_list):
508                if isinstance(statement, list):
509                    for idx, a in enumerate(statement):
510                        disjunct_list[d_idx][idx] = refine_smt_statement(a, self.y_name)
511                else:
512                    disjunct_list[d_idx] = refine_smt_statement(statement, self.y_name)
513
514        # Warning without reasons. self.x, self.y ARE lists but PyCharm is not convinced
515        self.in_coef_mat = self.__get_coef_mat(self.x, self.x_name, x_assert)
516        self.in_bias_mat = self.__get_bias_mat(x_assert)
517
518        if len(disjunct_list) == 0:  # If no 'OR' statements are present, just proceed normally
519            self.out_coef_mat.append(self.__get_coef_mat(self.y, self.y_name, y_assert))
520            self.out_bias_mat.append(self.__get_bias_mat(y_assert))
521        else:  # Otherwise, separate in different matrices
522            disjunct_list.reverse()
523            for d in disjunct_list:
524                if isinstance(d, list):
525                    self.out_coef_mat.append(self.__get_coef_mat(self.y, self.y_name, d))
526                    self.out_bias_mat.append(self.__get_bias_mat(d))
527                else:
528                    self.out_coef_mat.append(self.__get_coef_mat(self.y, self.y_name, [d]))
529                    self.out_bias_mat.append(self.__get_bias_mat([d]))
530
531        return self.in_coef_mat, self.in_bias_mat, self.out_coef_mat, self.out_bias_mat

555def read_smt_num(val: str):
556    """
557    Procedure to convert a SMTLIB string to a number.
558
559    Parameters
560    ----------
561    val: str
562        A string containing a number from a SMT file.
563
564    Returns
565    ----------
566    Any

568        float if it represents a float or
569        None if it does not represent a number.

571    """
572
573    try:

582        return

585def prefix2infix(prefix: str) -> str:
586    """
587    Procedure for converting a prefix string to an infix string.
588
589    Parameters
590    ----------
591    prefix: str
592        The prefix string that should be converted.
593
594    Returns
595    ----------
596    str
597        The infix-converted string.
598
599    """
600
601    # Preprocess
602    prefix = prefix.replace('(', ' ').replace(')', ' ').replace('assert', '').split()
603    stack = []
604
605    # Read tokens in reverse order
606    i = len(prefix) - 1
607    while i >= 0:
608        # Symbol is operand
609        if not is_operator(prefix[i]):
610            stack.append(prefix[i])
611            i -= 1
612        # Symbol is operator
613        else:
614            res = '(' + stack.pop() + ' ' + prefix[i] + ' ' + stack.pop() + ')'
615            stack.append(res)
616            i -= 1
617
618    return stack.pop()

621def refine_smt_statement(assertion: str, vec_name: str) -> str:
622    """
623    This method refines a SMTLIB statement by intercepting
624    malformed Normal Form formulas and rewriting them
625    correctly.
626
627    Parameters
628    ----------
629    assertion
630        The SMT formula to verify.
631    vec_name
632        The variable name in use
633
634    Returns
635    -------
636    str
637        The formula in input if it is already well-formed,
638        its refinement otherwise.
639
640    """
641
642    tree_converter = ExpressionTreeConverter()
643    tree = tree_converter.build_from_infix(assertion)
644
645    # Case Xi REL_OP Xj -> Xi - Xj REL_OP 0
646    # This is skipped if the right node is leaf
647    if read_smt_num(tree.node_right.data) is None:
648        assertion = '((' + tree.node_left.as_infix() + ' - ' + \
649                    tree.node_right.as_infix() + ') ' + tree.data + ' 0.0)'
650
651    # Case beta REL_OP Xi -> (-1.0 * Xi) REL_OP eval(-beta)
652    if read_smt_num(tree.node_left.data) is not None:
653        sign = '-'
654        if read_smt_num(tree.node_left.data) <= 0:
655            sign = ''
656        assertion = '((-1.0 * ' + tree.node_right.as_infix() + ') ' + \
657                    tree.data + f" {sign}" + tree.node_left.as_infix().replace('-', '') + ')'
658
659    # Case (Xi + alpha) REL_OP beta -> Xi REL_OP eval(beta - alpha)
660    # This is always performed
661    line = assertion.replace('(', ' ( ').replace(')', ' ) ').split()
662    idx = []
663    acc = 0
664
665    # Get variables index
666    for elem in line:
667        if vec_name in elem:
668            idx.append(line.index(elem))
669
670    for i in idx:
671        before = line[i - 1]
672        if before == '(':
673            # Try to evaluate the token as a number
674            post = read_smt_num(line[i + 2])
675            if post is not None:
676                if line[i + 1] == '+':
677                    acc += post
678                    line.pop(i + 1)
679                    line.pop(i + 1)
680                elif line[i + 1] == '-':
681                    acc -= post
682                    line.pop(i + 1)
683                    line.pop(i + 1)
684        elif before == '+':
685            post = read_smt_num(line[i - 2])
686            if post is not None:
687                acc += post
688                line.pop(i - 1)
689                line.pop(i - 1)
690        elif before == '-':
691            post = read_smt_num(line[i - 2])
692            if post is not None:
693                acc -= post
694                line.pop(i - 1)
695                line.pop(i - 1)
696
697    # If extra bias coefficients have been found, sum up the right hand side
698    if acc > 0:
699        relop_idx = line.index('<=')
700        line[relop_idx + 1] = str(read_smt_num(line[relop_idx + 1]) - acc)
701
702    # Save assertion
703    assertion = ''
704    for elem in line:
705        assertion += elem + ' '
706
707    return assertion.replace('( ', '(').replace(' )', ')')

312

314        ----------
315        vector: list

317        vec_name: str

329

331        n_var = len(vector)
332        coef_mat = np.zeros((len(asserts), n_var))
333
334        # Row counter
335        loop = 0
336
337        for a in asserts:
338            # Tokenize
339            line = a.replace('(', ' ( ').replace(')', ' ) ').split()
340            row = np.zeros(n_var)
341            idx = []
342
343            # Get variables index
344            for elem in line:
345                if vec_name in elem:
346                    idx.append(line.index(elem))
347
348            # Get coefficients for each index
349            for i in idx:

351                v_num = int(line[i].replace(vec_name + '_', ''))
352                before = line[i - 1]
353
354                # var on lhs
355                if before == '(':
356                    # Check if the coefficient is on the right side
357                    if line[i + 1] == '*':
358                        row[v_num] += read_smt_num(line[i + 2])
359                    elif line[i + 1] == '/':
360                        row[v_num] += float(1 / read_smt_num(line[i + 2]))
361                    else:
362                        row[v_num] += 1
363
364                # var as addendum (c = 1)
365                elif before == '+':
366                    row[v_num] += 1
367
368                # Same, but negative (c = -1)
369                elif before == '-':
370                    row[v_num] -= 1
371
372                # var with coefficient
373                elif before == '*':
374                    row[v_num] += read_smt_num(line[i - 2])
375
376            # Row now is complete and can be put in the matrix
377            coef_mat[loop] = row
378            loop += 1
379

def refine_smt_statement(assertion: str, vec_name: str) -> str: 
622def refine_smt_statement(assertion: str, vec_name: str) -> str:
623    """
624    This method refines a SMTLIB statement by intercepting
625    malformed Normal Form formulas and rewriting them
626    correctly.
627
628    Parameters
629    ----------
630    assertion
631        The SMT formula to verify.
632    vec_name
633        The variable name in use
634
635    Returns
636    -------
637    str
638        The formula in input if it is already well-formed,
639        its refinement otherwise.
640
641    """
642
643    tree_converter = ExpressionTreeConverter()
644    tree = tree_converter.build_from_infix(assertion)
645
646    # Case Xi REL_OP Xj -> Xi - Xj REL_OP 0
647    # This is skipped if the right node is leaf
648    if read_smt_num(tree.node_right.data) is None:
649        assertion = '((' + tree.node_left.as_infix() + ' - ' + \
650                    tree.node_right.as_infix() + ') ' + tree.data + ' 0.0)'
651
652    # Case beta REL_OP Xi -> (-1.0 * Xi) REL_OP eval(-beta)
653    if read_smt_num(tree.node_left.data) is not None:
654        sign = '-'
655        if read_smt_num(tree.node_left.data) <= 0:
656            sign = ''
657        assertion = '((-1.0 * ' + tree.node_right.as_infix() + ') ' + \
658                    tree.data + f" {sign}" + tree.node_left.as_infix().replace('-', '') + ')'
659
660    # Case (Xi + alpha) REL_OP beta -> Xi REL_OP eval(beta - alpha)
661    # This is always performed
662    line = assertion.replace('(', ' ( ').replace(')', ' ) ').split()
663    idx = []
664    acc = 0
665
666    # Get variables index
667    for elem in line:
668        if vec_name in elem:
669            idx.append(line.index(elem))
670
671    for i in idx:
672        before = line[i - 1]
673        if before == '(':
674            # Try to evaluate the token as a number
675            post = read_smt_num(line[i + 2])
676            if post is not None:
677                if line[i + 1] == '+':
678                    acc += post
679                    line.pop(i + 1)
680                    line.pop(i + 1)
681                elif line[i + 1] == '-':
682                    acc -= post
683                    line.pop(i + 1)
684                    line.pop(i + 1)
685        elif before == '+':
686            post = read_smt_num(line[i - 2])
687            if post is not None:
688                acc += post
689                line.pop(i - 1)
690                line.pop(i - 1)
691        elif before == '-':
692            post = read_smt_num(line[i - 2])
693            if post is not None:
694                acc -= post
695                line.pop(i - 1)
696                line.pop(i - 1)
697
698    # If extra bias coefficients have been found, sum up the right hand side
699    if acc > 0:
700        relop_idx = line.index('<=')
701        line[relop_idx + 1] = str(read_smt_num(line[relop_idx + 1]) - acc)
702
703    # Save assertion
704    assertion = ''
705    for elem in line:
706        assertion += elem + ' '
707
708    return assertion.replace('( ', '(').replace(' )', ')')
